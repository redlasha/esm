es modules

내가 사용하려고 만든 es module들 아래와 같이 사용할 수 있음 

        
        코드 예시

        <script type="module">
            import {ClickOnceButton} from 'https://redlasha.github.io/esm/click-once-button/click-once-button.js'
            let button = new ClickOnceButton();
            button.text = '테스트버튼'
            button.clickonce = () => {
                alert('test')
            }
            button.buttonstyle = 'color: red;';
            document.body.appendChild(button)
        </script>
